    def sample(self,key, shape=(1,)):

        samples = []
        keys = jax.random.split(key, len(self.priors))
        for (key,prior) in zip(keys,self.priors):
            samples.append(prior.sample(key,shape))
        return jnp.array(samples).T

    def logprior(self, values):
